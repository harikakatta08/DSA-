class LargestNumber {
    public static void main(String[] args) {
        int arr[]={1,54,221,2,5};
        int n=arr.length;
        int max=arr[0];
        for(int i=0;i<n-1;i++){
            if(arr[i]>max){
                max=arr[i];
            }
        }
        System.out.println(max);
    }
}
